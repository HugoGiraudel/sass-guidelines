{% highlight scss %} /// Mixin ajudando a definir ambos o `width` e o `height` simultaneamente. /// /// @author Hugo Giraudel /// /// @access public /// /// @param {Length} $width - `width` do elemento /// @param {Length} $height ($width) - `height` do elemento /// /// @example scss - Uso /// .foo { /// @include size(10em); /// } /// /// .bar { /// @include size(100%, 10em); /// } /// /// @example css - CSS gerado /// .foo { /// width: 10em; /// height: 10em; /// } /// /// .bar { /// width: 100%; /// height: 10em; /// } @mixin size($width, $height: $width) { width: $width; height: $height; } {% endhighlight %}
