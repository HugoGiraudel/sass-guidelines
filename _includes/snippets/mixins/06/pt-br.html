{% highlight scss %} /// Mixin auxiliar para colocar prefixos de terceiros /// @access public /// @author HugoGiraudel /// @param {String} $property - Propriedade CSS sem prefixo /// @param {*} $value - Valor puro de CSS /// @param {List} $prefixes - Lista de prefixos a serem gerados @mixin prefix($property, $value, $prefixes: ()) { @each $prefix in $prefixes { -#{$prefix}-#{$property}: $value; } #{$property}: $value; } {% endhighlight %}
