{% highlight scss %} /// Map de Z-indexes, coletando todos as camadas Z de uma aplicação /// @access private /// @type Map /// @prop {String} key - Nome da camada /// @prop {Number} value - Valor Z mapeado até a chave $z-indexes: ( 'modal': 5000, 'dropdown': 4000, 'default': 1, 'below': -1, ); /// Obtenha um valor de z-index a partir do nome de uma camada /// @access public /// @param {String} $layer - Nome da camada /// @return {Number} /// @require $z-indexes @function z($layer) { @if not map-has-key($z-indexes, $layer) { @error 'Não há nenhuma camada chamada `#{$layer}` em $z-indexes.
